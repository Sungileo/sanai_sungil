예시 데이터

name <- c("david","johns","yoon","michael")
math <- c(90,85,70,80)
art <- c(60,70,80,90)

ex <- data.frame(name,math,art)
ex
     name math art
1   david   90  60
2   johns   85  70
3    yoon   70  80
4 michael   80  90
ex$Total <- ex$math + ex$art
ex
     name math art Total
1   david   90  60   150
2   johns   85  70   155
3    yoon   70  80   150
4 michael   80  90   170
